;FIND NUMBER OF 1 BIT IN A BINARY NUMBER OF 16 BITS 
  

.MODEL SMALL
.STACK 100H
.CODE

MAIN PROC 
    MOV BX,1111000011101000B 
    MOV CX,16
    MOV AX,0
    
  TOP:
    ROL BX,1
    JNC NEXT
    INC AX
    
  NEXT: 
    LOOP TOP
    
    
    MOV AH,2
    ADD AX,48
    MOV DX,AX
    INT 21H
    
    EXIT:
    MOV AH,4CH
    INT 21H
    MAIN ENDP
END MAIN
